async function loginUser(emailUser, passwordUser) {
    const dataLogin = {
        "email": emailUser,
        "password": passwordUser
    };

    // Effectuer la requête fetch
    fetch('http://localhost:5678/api/users/login', {
        method: "POST",
        headers: { "Content-type": "application/json" },
        body: JSON.stringify(dataLogin)
    })
    .then(response => {
        console.log(response); // Affiche la réponse de l'API à la requête
    })
    .then(data => {
        data = response.json(); // Récupère la réponse en format JSON
        console.log(data);  // Affiche l'id et le token OU le message user not found
        return data;
    })
    .catch(error => {
        console.log('Erreur de requête:', error);
        window.alert("Erreur dans l’identifiant ou le mot de passe");
    });
}
